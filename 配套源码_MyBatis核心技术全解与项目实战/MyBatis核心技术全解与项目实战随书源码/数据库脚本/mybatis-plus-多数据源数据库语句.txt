-- 创建第一个库
DROP DATABASE IF EXISTS mybatisplusdsdb1;
CREATE DATABASE mybatisplusdsdb1;
use mybatisplusdsdb1;

-- 创建公民表
DROP TABLE IF EXISTS t_person;
CREATE TABLE t_person(
    p_id BIGINT(20) PRIMARY KEY AUTO_INCREMENT,
    p_name VARCHAR(30),
	p_gender VARCHAR(10),
    p_age INT(10)
);

-- 向公民表插入数据
INSERT INTO t_person (p_name, p_gender, p_age) VALUES
( 'lucy', 'female', 18),
( 'pipi', 'female', 19),
( 'dodo', 'female', 22);

-- 查询公民表中的数据
SELECT * FROM t_person;
